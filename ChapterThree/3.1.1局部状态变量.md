### 3.1.1局部状态变量
下面是创建银行对象一个过程；
```javascript
  function makeAccount(balance) {
    function withdraw(amount) {
      if (balance >= amount) {
        balance = balance - amount;
        return balance;
      } else if {
        return "Insufficient funds";
      }
    }
    function deposit(amount) {
      balance = balance + amount;
      return balance;
    } 
    return function dispatch(m) {
      if(m === "withdraw") {
        return withdraw;
      } else if (m=== "deposit") {
        return deposit;
      } else {
        return "Unknow request";
      }
      
    }
  }
  var acc = makeAccount(100);
  acc("withdraw")(10); //90;
  acc("deposit")(100); //190;
```
